\section{CHƯƠNG 1. TỔNG QUAN ĐỀ TÀI \& CÔNG NGHỆ}
% ================================================================

\subsection{Bối cảnh và lý do chọn đề tài}

Trong thời đại thương mại điện tử phát triển mạnh mẽ, nhu cầu mua sắm trực tuyến ngày càng tăng cao, kể cả với các mặt hàng đặc thù như cây cảnh và cây xanh trang trí. Xu hướng "sống xanh", làm việc tại nhà và trang trí không gian bằng cây xanh đã tạo ra một thị trường tiêu dùng sôi động, đặc biệt tại các đô thị lớn như TP.HCM và Hà Nội.

Tuy nhiên, hầu hết các cửa hàng cây cảnh vẫn hoạt động theo mô hình truyền thống, chưa có một nền tảng bán hàng trực tuyến chuyên nghiệp, hỗ trợ đầy đủ quy trình từ xem sản phẩm, đặt hàng, quản lý kho đến xử lý đơn hàng và thanh toán.

Từ bối cảnh đó, nhóm quyết định xây dựng \textbf{GreenSpace} --- một website thương mại điện tử chuyên bán cây cảnh, được thiết kế với giao diện hiện đại, quy trình mua hàng thuận tiện và hệ thống quản trị đầy đủ chức năng. Đề tài vừa giải quyết một bài toán thực tế, vừa là cơ hội để nhóm áp dụng toàn diện kiến thức lập trình web đã học, từ thiết kế CSDL, lập trình backend PHP đến xây dựng giao diện người dùng responsive.

\subsection{Mục tiêu}

\textbf{Mục tiêu cho người mua (end-user):}
\begin{itemize}
    \item Xem danh sách sản phẩm theo danh mục, tìm kiếm và lọc theo giá, sắp xếp đa chiều.
    \item Xem chi tiết sản phẩm: mô tả, hình ảnh gallery, thông tin chăm sóc.
    \item Quản lý giỏ hàng (thêm, cập nhật số lượng, xóa, xem tóm tắt).
    \item Đặt hàng với nhiều phương thức thanh toán (COD, chuyển khoản giả lập).
    \item Quản lý sổ địa chỉ giao hàng, chọn địa chỉ khi checkout.
    \item Xem lịch sử đơn hàng, theo dõi trạng thái, xem chi tiết và xác nhận thanh toán QR.
    \item Đăng ký, đăng nhập, cập nhật hồ sơ cá nhân.
    \item Gửi liên hệ / yêu cầu hỗ trợ qua form.
\end{itemize}

\textbf{Mục tiêu cho quản trị (admin):}
\begin{itemize}
    \item Đăng nhập qua URL riêng (\texttt{/admin/login}) với phân quyền chi tiết.
    \item Quản lý người dùng: tạo, sửa, khóa, phân quyền admin.
    \item Quản lý danh mục và sản phẩm: thêm/sửa/xóa mềm/ẩn hiện, upload ảnh.
    \item Nhập hàng theo lô (inventory batch), hỗ trợ xuất kho FIFO.
    \item Quản lý đơn hàng: cập nhật trạng thái luồng xử lý, duyệt/từ chối thanh toán chuyển khoản giả lập.
    \item Dashboard tổng hợp: thống kê doanh thu, đơn hàng, tồn kho thấp, khách hàng.
    \item Quản lý liên hệ từ khách hàng.
\end{itemize}

\subsection{Phạm vi thực hiện}

\textbf{Phạm vi chức năng đã thực hiện:}
\begin{itemize}
    \item Toàn bộ luồng end-user: xem sản phẩm, giỏ hàng, checkout, theo dõi đơn.
    \item Hệ thống quản trị: sản phẩm, danh mục, đơn hàng, người dùng, nhập kho, liên hệ.
    \item Thanh toán: COD và chuyển khoản giả lập (online\_mock) với quy trình duyệt admin.
    \item Quản lý kho theo lô (FIFO) với bảng \texttt{inventory\_batches} và \texttt{inventory\_batch\_allocations}.
    \item Phân quyền admin chi tiết (quyền quản lý sản phẩm, đơn hàng, người dùng, liên hệ, kho).
    \item Dashboard analytics: biểu đồ doanh thu, tồn kho thấp, top sản phẩm, top khách hàng.
\end{itemize}

\textbf{Phạm vi không thực hiện:}
\begin{itemize}
    \item Tích hợp cổng thanh toán thực (MoMo, ZaloPay, VNPay).
    \item Gửi email thông báo tự động.
    \item Chatbot tư vấn khách hàng.
    \item Hệ thống đánh giá sản phẩm công khai (đã có bảng CSDL, chưa xây dựng UI).
\end{itemize}

\subsection{Phân tích hệ thống}

\subsubsection{Đối tượng sử dụng}

Hệ thống GreenSpace phục vụ ba nhóm đối tượng chính:

\begin{enumerate}
    \item \textbf{Khách (Guest):} Người dùng chưa đăng nhập. Có thể xem danh sách sản phẩm theo danh mục, xem chi tiết sản phẩm, tìm kiếm và lọc, xem trang chăm sóc cây, gửi liên hệ. Không thể thêm giỏ hàng hay đặt hàng.

    \item \textbf{Khách hàng (Customer):} Người dùng đã đăng ký và đăng nhập với role \texttt{user}. Có đầy đủ quyền của Guest, thêm vào đó: quản lý giỏ hàng, đặt hàng, xem lịch sử đơn hàng, theo dõi trạng thái, xác nhận thanh toán QR, quản lý hồ sơ và sổ địa chỉ.

    \item \textbf{Quản trị viên (Admin):} Người dùng với role \texttt{admin} và tập quyền phù hợp. Truy cập qua URL \texttt{/admin/login} riêng biệt. Có quyền quản lý toàn bộ hệ thống tuỳ theo \texttt{admin\_permissions} được cấp: quản lý sản phẩm, danh mục, đơn hàng, người dùng, kho, liên hệ. Admin với quyền \texttt{admin.full\_access} có toàn quyền.
\end{enumerate}

\subsubsection{Quy tắc dữ liệu quan trọng}

\textbf{Quy tắc ẩn/xóa sản phẩm và danh mục:}
Hệ thống áp dụng chiến lược "xóa mềm" (soft delete). Khi admin yêu cầu xóa:
\begin{itemize}
    \item Nếu sản phẩm có dữ liệu trong \texttt{order\_details}: chuyển \texttt{status='inactive'} (không xóa khỏi DB để giữ lịch sử đơn hàng).
    \item Nếu danh mục có sản phẩm liên kết: chuyển \texttt{status='inactive'}.
    \item Chỉ xóa hẳn (DELETE) khi không có ràng buộc lịch sử.
\end{itemize}

\textbf{Quy tắc nhập hàng và xuất kho FIFO:}
\begin{itemize}
    \item Nhập: Tạo bản ghi mới trong \texttt{inventory\_batches} $\to$ \texttt{syncProductStock()} cập nhật \texttt{products.stock = SUM(quantity\_remaining)}.
    \item Xuất (khi tạo đơn): \texttt{InventoryBatch::deductFifo()} lấy các batch có \texttt{quantity\_remaining > 0} ORDER BY \texttt{received\_at ASC, id ASC} $\to$ trừ dần từ lô cũ nhất $\to$ ghi vào \texttt{inventory\_batch\_allocations}.
    \item Hoàn kho (khi hủy đơn): \texttt{restoreFifo()} đọc \texttt{inventory\_batch\_allocations} $\to$ cộng lại đúng batch đã trừ $\to$ xóa allocation records $\to$ \texttt{syncProductStock()}.
\end{itemize}

\textbf{Quy tắc tạo đơn hàng:}
\begin{itemize}
    \item Toàn bộ trong 1 transaction (BEGIN $\to$ lock rows FOR UPDATE $\to$ kiểm tra stock $\to$ tạo orders $\to$ tạo order\_details $\to$ trừ kho FIFO $\to$ tạo payments $\to$ COMMIT).
    \item Nếu bất kỳ bước nào thất bại: ROLLBACK, giỏ hàng không bị xóa.
    \item Phí ship: 0đ nếu subtotal $\geq$ 500.000đ, ngược lại 30.000đ.
    \item Đơn COD pending quá thời gian cấu hình: tự động hủy và hoàn kho.
\end{itemize}

\textbf{Quy tắc hết hạn đơn chờ:}
Mỗi request gọi \texttt{Order::expireStalePendingOrdersIfNeeded()} (chỉ chạy 1 lần/request qua static flag). Đơn \texttt{order\_status='pending'} và \texttt{payment\_status='unpaid'} quá hạn sẽ bị chuyển sang \texttt{order\_status='cancelled'} và hoàn kho.

\subsubsection{Thiết kế module}

\textbf{Module Auth:}
\begin{itemize}
    \item \textbf{Chức năng:} quản lý đăng ký, đăng nhập, đăng xuất cho khách hàng và luồng đăng nhập tách riêng cho admin. Module này là điểm vào đầu tiên của toàn bộ cơ chế xác thực và tạo session người dùng.
    \item \textbf{Input/Output:} input của login gồm \texttt{identifier}, \texttt{password}, \texttt{csrf\_token}, \texttt{redirect}; input của signup gồm \texttt{full\_name}, \texttt{username}, \texttt{email}, \texttt{phone}, \texttt{password}, \texttt{confirm\_password}. Output là \texttt{Response} redirect về trang đích an toàn hoặc render lại form cùng tập lỗi inline; ở mức service, output là mảng \texttt{success/errors/user\_id}.
    \item \textbf{Controller/Service/Model liên quan:} \texttt{AuthController} xử lý login/signup/logout storefront; \texttt{AdminAuthController} xử lý login admin; \texttt{AuthService} thực hiện \texttt{login()}, \texttt{loginAdmin()}, \texttt{register()}; model chính là \texttt{User}. Ngoài ra module còn dùng helper CSRF, session và \texttt{safe\_redirect\_target()}.
    \item \textbf{Luồng xử lý chính:} controller nhận request $\to$ kiểm tra rate limit và CSRF $\to$ chuẩn hóa dữ liệu nhập $\to$ gọi \texttt{AuthService} $\to$ service kiểm tra tài khoản bằng \texttt{User::findByLogin()}, xác minh mật khẩu bằng bcrypt, áp dụng dummy hash khi tài khoản không tồn tại $\to$ tạo session đăng nhập và redirect. Với signup, service kiểm tra regex username, uniqueness của username/email, băm mật khẩu rồi insert bản ghi mới vào \texttt{users}.
\end{itemize}

\textbf{Module Catalog:}
\begin{itemize}
    \item \textbf{Chức năng:} hiển thị danh sách sản phẩm, chi tiết sản phẩm, danh mục và tìm kiếm/lọc/sắp xếp theo nhiều điều kiện mà vẫn giữ tương thích với cả route MVC mới lẫn các trang public cũ.
    \item \textbf{Input/Output:} input của module là \texttt{slug} sản phẩm, \texttt{slug} danh mục và các query param như \texttt{search}, \texttt{category}, \texttt{sort}, \texttt{page}, \texttt{min\_price}, \texttt{max\_price}. Output là view model hoàn chỉnh gồm danh sách sản phẩm, sản phẩm chi tiết, danh mục, metadata phân trang, URL state cho filter và response fragment khi request AJAX.
    \item \textbf{Controller/Service/Model liên quan:} \texttt{ProductController} cung cấp \texttt{index()}, \texttt{detail()}, \texttt{detailByRequest()}, \texttt{category()}; \texttt{ProductCatalogRequestData} gom input typed; \texttt{ProductService} truy vấn nghiệp vụ; \texttt{ProductPresenter} chuẩn bị dữ liệu cho view. Model chính là \texttt{Product} và \texttt{Category}.
    \item \textbf{Luồng xử lý chính:} request danh sách đi vào \texttt{ProductController} $\to$ DTO chuẩn hóa filter $\to$ \texttt{ProductPresenter::presentCatalog()} gọi \texttt{ProductService::getProducts()} và \texttt{validateSort()} $\to$ service truy vấn DB, tính tổng bản ghi, phân trang, lấy category hiện tại $\to$ presenter đóng gói closures tạo URL lọc $\to$ controller quyết định render full page hay chỉ \texttt{storefront/products/content} nếu là AJAX. Với chi tiết sản phẩm, controller nhận \texttt{slug} $\to$ presenter gọi \texttt{getProductDetail()} $\to$ nếu không có dữ liệu thì trả 404, ngược lại render trang detail.
\end{itemize}

\textbf{Module Cart \& Checkout:}
\begin{itemize}
    \item \textbf{Chức năng:} quản lý giỏ hàng theo session, cập nhật số lượng theo thời gian thực, hiển thị tóm tắt đơn và tạo đơn hàng hoàn chỉnh từ giỏ kèm địa chỉ giao hàng, phương thức thanh toán và trừ kho FIFO.
    \item \textbf{Input/Output:} input của cart là \texttt{action}, \texttt{product\_id}, \texttt{quantity}, \texttt{csrf\_token}, cờ \texttt{ajax}; input của checkout gồm \texttt{selected\_address\_id}, thông tin người nhận, địa chỉ, \texttt{payment\_method}, \texttt{note}, \texttt{save\_as\_default}. Output của cart là JSON hoặc redirect với \texttt{cart\_count}/message; output của checkout là \texttt{order\_id}, lỗi validate, draft dữ liệu địa chỉ hoặc redirect sang trang đơn hàng.
    \item \textbf{Controller/Service/Model liên quan:} \texttt{CartController} xử lý hiển thị và mutation giỏ; \texttt{CheckoutController} xử lý trang thanh toán; \texttt{CartService} quản lý \texttt{\$\_SESSION['cart']}; \texttt{CheckoutService} thực hiện \texttt{placeOrder()}; \texttt{StorefrontPagePresenter} dựng dữ liệu checkout. Model chính gồm \texttt{Product}, \texttt{Order}, \texttt{Address}, \texttt{User}; ở lớp DB còn tương tác với \texttt{payments}, \texttt{inventory\_batches} và \texttt{inventory\_batch\_allocations}.
    \item \textbf{Luồng xử lý chính:} người dùng thêm/cập nhật/xóa giỏ $\to$ \texttt{CartController} kiểm tra CSRF và kiểu request $\to$ gọi page service hoặc \texttt{CartService} để mutate session $\to$ hydrate lại giá/tồn kho thật từ DB trước khi trả response. Khi checkout, \texttt{CheckoutController} gom toàn bộ POST $\to$ \texttt{CheckoutService::placeOrder()} validate người nhận và địa chỉ $\to$ khóa sản phẩm bằng \texttt{FOR UPDATE}, tính summary, tạo đơn/payout record trong transaction $\to$ reserve inventory theo FIFO $\to$ commit và xóa session cart. Nếu chọn \texttt{online\_mock}, hệ thống giữ trạng thái \texttt{unpaid} và điều hướng người dùng đến cụm xác nhận QR.
\end{itemize}

\begin{itemize}

\subsubsection{Các truy vấn SQL tiêu biểu}

\textbf{Truy vấn 1 -- Phân trang danh sách sản phẩm với lọc đa điều kiện:}

\begin{lstlisting}[language=SQL, caption=Truy vấn phân trang sản phẩm có lọc]
SELECT p.*, c.name AS category_name, c.slug AS category_slug,
       CASE WHEN p.sale_price IS NOT NULL AND p.sale_price > 0
            THEN p.sale_price ELSE p.price END AS effective_price
FROM products p
LEFT JOIN categories c ON p.category_id = c.id
WHERE p.status = 'active'
  AND p.category_id = :category_id      -- nếu lọc danh mục
  AND (CASE WHEN p.sale_price IS NOT NULL AND p.sale_price > 0
            THEN p.sale_price ELSE p.price END) BETWEEN :min_price AND :max_price
  AND (p.name LIKE :search_name OR p.description LIKE :search_description)
ORDER BY p.featured DESC, p.created_at DESC
LIMIT :limit OFFSET :offset;
-- Mục đích: Lấy sản phẩm active có lọc giá effective, danh mục, từ khóa
-- Input: category_id, min/max price, search keyword, page/limit
-- Output: Danh sách sản phẩm đã phân trang
\end{lstlisting}

\textbf{Truy vấn 2 -- Lấy tổng sản phẩm cho phân trang:}

\begin{lstlisting}[language=SQL, caption=Đếm tổng sản phẩm cho phân trang]
SELECT COUNT(*) AS total
FROM products p
LEFT JOIN categories c ON p.category_id = c.id
WHERE p.status = 'active'
  AND (các điều kiện lọc tương tự truy vấn 1);
-- Mục đích: Tính tổng số trang (total_pages = CEIL(total / limit))
\end{lstlisting}

\textbf{Truy vấn 3 -- Tạo đơn hàng (Transaction):}

\begin{lstlisting}[language=SQL, caption=Transaction tạo đơn hàng]
-- Bước 1: Lock sản phẩm
SELECT id, stock FROM products WHERE id = :product_id LIMIT 1 FOR UPDATE;

-- Bước 2: Tạo đơn hàng
INSERT INTO orders (user_id, order_number, full_name, ..., total_amount, 
                    payment_method, payment_status, order_status)
VALUES (:user_id, :order_number, ..., :total_amount, 'cod', 'unpaid', 'pending');

-- Bước 3: Tạo chi tiết đơn
INSERT INTO order_details (order_id, product_id, product_name, price, 
                           quantity, subtotal)
VALUES (:order_id, :product_id, :name, :price, :qty, :subtotal);

-- Bước 4: Tạo payment
INSERT INTO payments (order_id, provider, status, amount)
VALUES (:order_id, 'cod', 'unpaid', :amount);
-- Mục đích: Tạo đơn hàng toàn vẹn trong 1 transaction
-- Input: Thông tin giỏ hàng, địa chỉ giao, PTTT
-- Output: order_id mới
\end{lstlisting}

\textbf{Truy vấn 4 -- Xuất kho theo FIFO:}

\begin{lstlisting}[language=SQL, caption=Xuất kho FIFO từ các lô]
-- Lấy các lô còn hàng, cũ nhất trước
SELECT * FROM inventory_batches
WHERE product_id = :product_id AND quantity_remaining > 0
ORDER BY received_at ASC, id ASC;

-- Trừ kho từng lô (lặp đến khi đủ số lượng)
UPDATE inventory_batches
SET quantity_remaining = quantity_remaining - :deduct
WHERE id = :batch_id AND quantity_remaining >= :deduct;

-- Ghi allocation để có thể hoàn kho chính xác
INSERT INTO inventory_batch_allocations 
       (order_id, product_id, batch_id, quantity)
VALUES (:order_id, :product_id, :batch_id, :deduct);
-- Mục đích: Xuất kho đúng FIFO, ghi nhận để hoàn kho chính xác
\end{lstlisting}

\textbf{Truy vấn 5 -- Đồng bộ tồn kho từ batch:}

\begin{lstlisting}[language=SQL, caption=Đồng bộ stock từ inventory\_batches]
UPDATE products 
SET stock = (
    SELECT COALESCE(SUM(quantity_remaining), 0)
    FROM inventory_batches
    WHERE product_id = :product_id
), updated_at = CURRENT_TIMESTAMP
WHERE id = :product_id;
-- Mục đích: Giữ products.stock luôn đồng bộ với tổng batch remaining
\end{lstlisting}

\textbf{Truy vấn 6 -- Dashboard: Top sản phẩm bán chạy:}

\begin{lstlisting}[language=SQL, caption=Thống kê top sản phẩm bán chạy]
SELECT p.id, p.name, p.stock,
       COALESCE(SUM(od.quantity), 0) AS units_sold,
       COALESCE(SUM(od.subtotal), 0) AS gross_revenue,
       COUNT(DISTINCT o.id) AS order_count,
       COUNT(DISTINCT o.user_id) AS customer_count
FROM products p
LEFT JOIN (
    SELECT od.product_id, od.quantity, od.subtotal, od.order_id
    FROM order_details od
    INNER JOIN orders o ON o.id = od.order_id
    WHERE o.order_status <> 'cancelled'
) od ON od.product_id = p.id
WHERE p.status = 'active'
GROUP BY p.id
ORDER BY units_sold DESC, gross_revenue DESC
LIMIT 6;
-- Mục đích: Tìm sản phẩm bán chạy nhất (không tính đơn hủy)
\end{lstlisting}

\textbf{Truy vấn 7 -- Báo cáo tồn kho sắp hết:}

\begin{lstlisting}[language=SQL, caption=Sản phẩm sắp hết hàng]
SELECT name, slug, stock, price, sale_price
FROM products
WHERE status = 'active' AND stock <= 5
ORDER BY stock ASC, updated_at DESC
LIMIT 8;
-- Mục đích: Cảnh báo sản phẩm tồn kho 1-5 cần nhập thêm
\end{lstlisting}

\textbf{Truy vấn 8 -- Biểu đồ doanh thu theo ngày:}

\begin{lstlisting}[language=SQL, caption=Doanh thu 7 ngày gần nhất]
SELECT DATE(created_at) AS bucket_key,
       COALESCE(SUM(total_amount), 0) AS revenue_total
FROM orders
WHERE created_at >= :start_date
  AND order_status <> 'cancelled'
  AND payment_status = 'paid'
GROUP BY bucket_key
ORDER BY bucket_key ASC;
-- Mục đích: Cung cấp dữ liệu cho biểu đồ doanh thu dashboard
-- Input: start_date (hôm nay - 6 ngày)
-- Output: Mảng (ngày, doanh thu) để render chart
\end{lstlisting}

\textbf{Truy vấn 9 -- Lấy số dư tồn kho đã trừ cho đơn hàng:}

\begin{lstlisting}[language=SQL, caption=Kiểm tra số lượng đã trừ kho cho đơn]
SELECT product_id,
       COALESCE(SUM(
           CASE WHEN action = 'deduct' THEN quantity
                WHEN action = 'restore' THEN -quantity
                ELSE 0 END
       ), 0) AS balance
FROM inventory_logs
WHERE order_id = :order_id
GROUP BY product_id;
-- Mục đích: Biết sản phẩm nào đã được trừ/hoàn kho cho đơn,
--           tránh trừ hoặc hoàn trùng
\end{lstlisting}

\textbf{Truy vấn 10 -- Hủy đơn hàng cũ tự động:}

\begin{lstlisting}[language=SQL, caption=Tìm đơn hàng quá hạn cần hủy tự động]
SELECT id, order_number, payment_method
FROM orders
WHERE order_status = 'pending'
  AND payment_status = 'unpaid'
  AND (
    (payment_method = 'online_mock' 
     AND created_at <= CURRENT_TIMESTAMP - INTERVAL 30 MINUTE)
    OR
    (payment_method = 'cod' 
     AND created_at <= CURRENT_TIMESTAMP - INTERVAL 1440 MINUTE)
  )
ORDER BY created_at ASC
FOR UPDATE;
-- Mục đích: Tự động hủy đơn pending quá thời gian cấu hình
--           (online_mock: 30 phút, COD: 24 giờ)
\end{lstlisting}

\subsection*{Kết quả đạt được}

Sau quá trình thực hiện, nhóm đã hoàn thành đồ án website bán cây cảnh \textbf{GreenSpace} với đầy đủ các chức năng theo yêu cầu:

\textbf{Checklist End-user:}
\begin{itemize}
    \item[$\checkmark$] Đăng ký / Đăng nhập / Đăng xuất với CSRF bảo vệ
    \item[$\checkmark$] Xem sản phẩm theo danh mục, tìm kiếm, lọc giá (AJAX)
    \item[$\checkmark$] Xem chi tiết sản phẩm với gallery ảnh, thông tin chăm sóc
    \item[$\checkmark$] Quản lý giỏ hàng (thêm/cập nhật/xóa) với animation fly-to-cart
    \item[$\checkmark$] Checkout với sổ địa chỉ, lưu nhanh địa chỉ
    \item[$\checkmark$] Thanh toán COD và chuyển khoản giả lập QR
    \item[$\checkmark$] Xem lịch sử đơn hàng, theo dõi trạng thái
    \item[$\checkmark$] Xác nhận thanh toán QR từ trang chi tiết đơn
    \item[$\checkmark$] Quản lý hồ sơ, sổ địa chỉ (CRUD)
    \item[$\checkmark$] Gửi liên hệ với validate đầy đủ
\end{itemize}

\textbf{Checklist Admin:}
\begin{itemize}
    \item[$\checkmark$] Đăng nhập admin URL riêng
    \item[$\checkmark$] Dashboard analytics với thống kê, top sản phẩm, top khách hàng, biểu đồ doanh thu
    \item[$\checkmark$] Quản lý sản phẩm: CRUD, upload ảnh, xóa mềm
    \item[$\checkmark$] Quản lý danh mục: CRUD, xóa mềm, danh mục cha
    \item[$\checkmark$] Nhập kho theo lô (FIFO) với bảng inventory\_batches
    \item[$\checkmark$] Quản lý đơn hàng: cập nhật trạng thái, duyệt/từ chối thanh toán
    \item[$\checkmark$] Quản lý người dùng: sửa, khóa, phân quyền admin chi tiết
    \item[$\checkmark$] Quản lý liên hệ: đọc, cập nhật trạng thái xử lý
    \item[$\checkmark$] Cảnh báo tồn kho thấp
    \item[$\checkmark$] Tự động hủy đơn hàng quá hạn
\end{itemize}

\subsection*{Hạn chế}

\begin{itemize}
    \item \textbf{Thời gian:} Một số chức năng nâng cao như wishlist UI, review UI chưa hoàn thiện mặc dù đã có bảng CSDL và dữ liệu mẫu.
    \item \textbf{Bảo mật:} Chưa có rate limiting cho API cart, chưa có CAPTCHA cho form liên hệ.
    \item \textbf{Hiệu năng:} Chưa có caching layer (Redis/Memcached) cho các truy vấn nặng như dashboard.
    \item \textbf{Giao diện:} Biểu đồ dashboard chưa render trực tiếp dạng chart (dữ liệu có sẵn nhưng chưa gắn với thư viện chart.js/apexcharts).
    \item \textbf{Thanh toán:} Chỉ mô phỏng, chưa tích hợp cổng thanh toán thực.
    \item \textbf{Email:} Chưa gửi email xác nhận đơn hàng tự động.
\end{itemize}

\subsection*{Hướng phát triển}

\begin{enumerate}
    \item \textbf{Thanh toán online thực:} Tích hợp MoMo, ZaloPay, VNPay thay thế online\_mock.
    \item \textbf{Email notification:} Gửi email xác nhận đơn hàng, thông báo trạng thái qua SMTP (PHPMailer).
    \item \textbf{Phân quyền chi tiết (RBAC):} Mở rộng hệ thống quyền hiện có, thêm quản lý role động.
    \item \textbf{Chatbot / Hỗ trợ khách hàng:} Tích hợp chatbot tư vấn chăm sóc cây hoặc live chat.
    \item \textbf{Review \& Rating:} Hoàn thiện UI cho hệ thống đánh giá sản phẩm (bảng CSDL đã có).
    \item \textbf{SEO \& Performance:} Server-side caching, lazy loading ảnh, sitemap.xml, meta tag động.
    \item \textbf{Mobile App:} Xây dựng REST API để phục vụ ứng dụng mobile (React Native / Flutter).
    \item \textbf{Analytics nâng cao:} Tích hợp Google Analytics, heatmap, A/B testing.
\end{enumerate}
